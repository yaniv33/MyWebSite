Declare HTTP Request

constructor(public http: HttpClient,                    // 1. inject dependency
return this.http.request('POST', 'URL',         // SPECIFY METHOD & URL
  {  body: parameter,                               // PLACE OBJECT IN BODY
            responseType: 'json',                           // SPECIFY RESPONSE TYPE
            observe: 'body',                                // HOW MUCH OF THE RESPONSE TO RETURN
            headers: { Authorization: `Bearer $uweoru`,     // HEADER
              timeout:       `${6000}`,
              index:          '0'}
  })                                                                      // THIS WILL RETURN OBSERVABLE
  .pipe(catchError(this.handleError<[]>("get hello", [])))    // HERE WE CAN PIPE AN ERROR
  .subscribe( {                                                     // THE SUBSCRIBE IS ALSO THE SEND REQUEST
    next: (value) => {                                        // NEXT : DEAL WITH THE RESPONSE
      console.log(value); 
    },                                          
    complete: () => { console.log('complete')},                       // COMPLETE : ON COMPLETE HOOK
    error: (error) => {                                               // ERROR : ON ERROR HOOK
      console.error('Problem Controller: ' + error.message);
    },                                          
  });

Work with arrays

[1,2,3,4,5].map(num => num * 10 );          // [10 20 30 40 50] 
[1,2,3,4,5].forEach(num => console.log('got ' + (num +2)));
[1,2,3,4,5].filter(num => num > 3)          // [4 5] 
[1,2,3,4,5].reduce((accumulator, currentValue) => accumulator + currentValue)  // 15 , accumulator get first value
[1,2,3,4,5].findIndex(num => num === 3);   //  3
[4,5,1,3,2].sort((a, b) => a - b);  // [1,2,3,4,5] if result negative, switch
[1,2,3,4,5].slice(1,4)    // [2,3,4]  list from start to end, if end negative count from end
[1,2,3,4,5].splice(from loc:1,count: 2 ,swap with 8,9) // return [2,3] origin change to [1,8,9,4,5]

[1,2,3,4,5].some(x => x > 2 )   // true
[1,2,3,4,5].every(x => x > 2 )  // false
[1,2,3,4,5].includes(3)         // true

REMARK – THE MARKED FUNCTIONS CHANGE THE ORIGIN OBJECT!!, in order to avoid use .slice()

Form management

import {FormBuilder,  FormGroup, Validators} from "@angular/forms";
myForm: FormGroup;
constructor(private formBuilder: FormBuilder) {}  // FormBuilder > FormGroup > Validators
ngOnInit() {
  this.myForm = this.formBuilder.group({
    name: ['Yaniv'],
    age: [null, [
      Validators.required,
      Validators.minLength(2),
      Validators.min(18),
      Validators.max(65)
    ]],
    password: ['', [
      Validators.required,
      Validators.pattern('[a-d]')
    ]],
    agree: [false, [
      Validators.requiredTrue
    ]],
  });
}
onSubmit(data) {
  const formValue = this.myForm.value;
  console.log(formValue.name);
  this.result = JSON.stringify(formValue);
}
<div [formGroup]="myForm" (ngSubmit)="onSubmit()">
  <input type="text"     formControlName="name" placeholder="name" ><br/>
  <input type="number"   formControlName="age" placeholder="age" appDigitsOnly inputmode="numeric"  ><br/>
  <input type="text"     formControlName="password" placeholder="password" ><br/>
  <input type="checkbox" formControlName="agree" > Agree<br/>
</div>
